 2function az = music_from_matlab(channel1,channel2,channel3,channel4)
 3
 4%% Asserts
 5assert(isa(channel1, 'double') && ~isreal(channel1) && all(size(channel1) == [4096,1]))
 6assert(isa(channel2, 'double') && ~isreal(channel2) && all(size(channel2) == [4096,1]))
 7assert(isa(channel3, 'double') && ~isreal(channel3) && all(size(channel3) == [4096,1]))
 8assert(isa(channel4, 'double') && ~isreal(channel4) && all(size(channel4) == [4096,1]))
 9
10%% Predefines
11persistent music
12if isempty(music)
13    fc = 2.437e9;
14    fs = 100e6/256;
15    elements = 4;
16    lambda = physconst('LightSpeed')/fc;
17    hula = phased.ULA('NumElements',elements,'ElementSpacing',lambda/2);
18    hula.Element.FrequencyRange = [fc-fs/2 fc+fs/2];
19    
20    music = phased.RootMUSICEstimator(...
21        'SensorArray',hula,...
22        'OperatingFrequency',fc,...
23        'NumSignalsSource','Property','NumSignals',1,...
24        'ForwardBackwardAveraging',true); 
25end
26
27rxsig = [channel1, channel2, channel3, channel4];
28
29%% Output
30doa_est = 1.0;
31doa_est(1) = step(music,rxsig);
32
33% Convert to azimuth
34az = 1.0;
35elevation = 0;
36az(1) = broadside2az(doa_est,elevation);
37
38end
